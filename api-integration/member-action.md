    project: Object,

  async created() {

  methods: {
    invite_member_action() {
      axios
        .post(
          process.env.VUE_APP_API +
            "/api_kanban_board/service/projects/invite_user.php",
          {
            id: this.project_id,
            email_array: this.invite_member
              .map((x) => x.email)
              .filter((x) => x != ""),
          },
          {
            headers: {
              authorization: localStorage.getItem("token"),

        )
        .then(async (res) => {
          if (res.data.status == true) {
